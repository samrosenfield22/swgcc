
<stmt>			::= <comma> ";"
<stmt>			::= <decl> ";"

<decl>			::= type id {push %s} <mdecl>?
<mdecl>			::= "=" <comma> {assign}

<comma>       ::= <assign> <mcomma>*
<mcomma>      ::= "," <assign>


<assign> ::= id <assign_id>
<assign> ::= num <mlogical>*
<assign> ::= "(" <comma> ")" <mlogical>*

<assign_id> ::= "=" <assign>*
<assign_id> ::= "+=" <assign>*
<assign_id> ::= <mlogical>*



<assign>      ::= id <massign>+
<assign>      ::= <logical>
<massign>     ::= "=" <assign>
<massign>     ::= "+=" <assign>
<massign>     ::= "-=" <assign>
<massign>     ::= "*=" <assign>
<massign>     ::= "/=" <assign>
<massign>     ::= "%=" <assign>
<massign>     ::= "<<=" <assign>
<massign>     ::= ">>=" <assign>
<massign>     ::= "&=" <assign>
<massign>     ::= "|=" <assign>
<massign>     ::= "^=" <assign>

<prod>        ::= <base> <mprod>*
<mprod>       ::= "*" <base> {mult}
<mprod>       ::= "/" <base> {div}
<mprod>       ::= "%" <base> {mod}

<base>        ::= num {push %d}
<base>        ::= id {push id}
<base>        ::= "(" <comma> ")"
